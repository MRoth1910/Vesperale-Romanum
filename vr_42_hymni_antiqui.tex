%\header{hymni antiqui.}
\cheader{hymni antiqui.}

\rubrique{Tonus cujusque Hymni infra positi semper idem est ac Tonus
ejusdem Hymni secundum usum novum, sicut ponitur in Antiphonario,
nisi aliter notetur.}

\rubrique{Ceterum, quæcumque præscribuntur in Antiphonario circa tonum
et doxologiam aliquando mutanda in Hymnis ratione Festi, vel Temporis aut Octavae currentis, 
vel circa vocales aut syllabas praeter metrum
occurrentes: omnino pariter observanda sunt in hoc Hymnario antiquo,
nisi aliter notetur.}

\rubrique{Hymni pro quibus non est textus antiquus, sumendi sunt ex Antiphonario.}
